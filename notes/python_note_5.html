def AUM_trans(s):
    patt='\d*\.\d*'
    m=re.search(patt,s)
    if s[-1]=='B':
        return float(m.group())*1000
    else:
        return float(m.group())

def plot_price2(data,price_type,start=None,end=None,sma=None,ewma=None):
    data = data.loc[start:end,[price_type]]
    data.plot(figsize=(15,5)).set(xlabel="Date", ylabel=price_type)
    legend=[price_type]
    if sma!=None:
        data['SMA'] = data[price_type].rolling(sma).mean()
        data['SMA'].plot(figsize=(15,5))
        data['SMA_expand'] = data[price_type].expanding(sma).mean()
        data['SMA_expand'].plot(figsize=(15,5))
        legend.append('SMA-Rolling')
        legend.append('SMA-Expanding')
    if ewma!=None:
        data['EWMA'] = data[price_type].ewm(span=ewma).mean()
        data['EWMA'].plot(figsize=(15,5))
        legend.append('EWMA')
        plt.legend(tuple(legend))
